-- 1) Create the tables
CREATE TABLE student_profile (
    rollno INT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    dob DATE NOT NULL,
    department VARCHAR(255) NOT NULL,
    city VARCHAR(255) NOT NULL
);

CREATE TABLE marks (
    rollno INT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    mark1 INT NOT NULL,
    mark2 INT NOT NULL
);
